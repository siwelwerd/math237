\[W=\mathrm{span}\ \left\{\left[\begin{array}{r}

-2 \\
2 \\

3
\end{array}\right] , \left[\begin{array}{r}

5 \\

-5 \\
-15
\end{array}\right] , \left[\begin{array}{r}
-2 \\
-4 \\

3 \\
-3
\end{array}\right] , \left[\begin{array}{r}
9 \\
17 \\
-17 \\
-9 \\

  \end{exerciseStatement}
