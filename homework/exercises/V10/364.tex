  4 x_ 1 + 1 x_ 2 -4 x_ 3 -6 x_ 4 -2 x_ 5 &= 0  \\ 
  -3 x_ 1 -4 x_ 2 -3 x_ 4 -5 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}
