\begin{align*}
 -1 x_ 1 + 4 x_ 2 + 5 x_ 3 + 3 x_ 4 -3 x_ 5 &= 0  \\ 
  3 x_ 1 + 4 x_ 2 -3 x_ 3 + 5 x_ 5 &= 0  \\ 
  5 x_ 1 -3 x_ 2 -3 x_ 3 -5 x_ 4 -4 x_ 5 &= 0  \\ 
  -5 x_ 1 + 2 x_ 2 + 5 x_ 3 -1 x_ 4 + 4 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

-\frac{551}{412} \\

\end{array}\right]\right\}.\]
