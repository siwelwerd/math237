\begin{align*}
 -5 x_ 1 -4 x_ 2 + 2 x_ 3 -6 x_ 4 + 4 x_ 5 &= 0  \\ 
  -6 x_ 1 -3 x_ 2 + 1 x_ 3 -3 x_ 4 -6 x_ 5 &= 0  \\ 
  5 x_ 1 + 3 x_ 2 -4 x_ 3 -1 x_ 4 -3 x_ 5 &= 0  \\ 
  -3 x_ 1 -5 x_ 2 + 3 x_ 3 -3 x_ 4 + 4 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

-\frac{35}{22} \\
-\frac{107}{44} \\
