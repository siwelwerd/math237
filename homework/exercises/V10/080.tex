\begin{align*}
 -12 x_ 1 + 4 x_ 2 + 4 x_ 3 -8 x_ 4 + 4 x_ 5 &= 0  \\ 
  8 x_ 1 -6 x_ 2 + 14 x_ 3 + 22 x_ 4 + 4 x_ 5 &= 0  \\ 
  -1 x_ 1 -1 x_ 2 + 7 x_ 3 + 6 x_ 4 + 3 x_ 5 &= 0  \\ 
  5 x_ 1 -1 x_ 2 -5 x_ 3 -3 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

  \end{exerciseAnswer}
