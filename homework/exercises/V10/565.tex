\begin{align*}
 -4 x_ 1 -8 x_ 2 -1 x_ 4 -4 x_ 5 &= 0  \\ 
  -3 x_ 1 -1 x_ 2 -5 x_ 3 -4 x_ 4 + 2 x_ 5 &= 0  \\ 
  4 x_ 1 + 12 x_ 2 -4 x_ 3 + 5 x_ 4 + 8 x_ 5 &= 0  \\ 
  1 x_ 1 + 1 x_ 2 + 1 x_ 3 + 3 x_ 4 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

\end{array}\right]\right\}.\]
