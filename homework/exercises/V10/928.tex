 -4 x_ 2 + 5 x_ 3 + 20 x_ 4 -5 x_ 5 &= 0  \\

  14 x_ 1 -1 x_ 2 -6 x_ 3 -3 x_ 4 -1 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}
