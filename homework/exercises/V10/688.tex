\begin{align*}
 -2 x_ 1 + 5 x_ 2 -6 x_ 3 -3 x_ 4 -1 x_ 5 &= 0  \\ 
  -10 x_ 1 + 1 x_ 2 -4 x_ 3 + 11 x_ 4 -5 x_ 5 &= 0  \\ 
  -12 x_ 1 -5 x_ 2 + 4 x_ 3 + 22 x_ 4 -6 x_ 5 &= 0  \\ 
  2 x_ 1 + 2 x_ 2 -3 x_ 3 -6 x_ 4 + 1 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}
