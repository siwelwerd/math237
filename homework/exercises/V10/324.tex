\begin{align*}
 -2 x_ 1 -5 x_ 2 + 4 x_ 3 + 5 x_ 4 -9 x_ 5 &= 0  \\ 
  + 1 x_ 2 -1 x_ 3 + 5 x_ 4 -5 x_ 5 &= 0  \\

  -5 x_ 1 -4 x_ 2 + 1 x_ 3 + 3 x_ 4 -13 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

  \begin{exerciseAnswer}
   A basis is   
\[\left\{\left[\begin{array}{c}
-2 \\
0 \\
0 \\
1 \\
1
\end{array}\right]\right\}.\]

  


  \end{exerciseAnswer}
