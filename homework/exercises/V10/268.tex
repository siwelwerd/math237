\begin{align*}

  5 x_ 1 -10 x_ 3 + 5 x_ 4 -3 x_ 5 &= 0  \\ 
  2 x_ 1 -8 x_ 2 -8 x_ 3 -2 x_ 4 + 5 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}
