  1 x_ 1 -4 x_ 2 -5 x_ 3 -6 x_ 5 &= 0  \\ 
  4 x_ 1 -4 x_ 2 + 5 x_ 4 -3 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}
