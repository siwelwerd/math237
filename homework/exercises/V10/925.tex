\begin{align*}
 -6 x_ 2 + 3 x_ 3 + 4 x_ 4 -1 x_ 5 &= 0  \\ 
  -1 x_ 1 -6 x_ 3 -3 x_ 4 + 5 x_ 5 &= 0  \\ 
  + 4 x_ 2 + 3 x_ 3 -6 x_ 4 + 3 x_ 5 &= 0  \\ 
  + 4 x_ 2 -4 x_ 3 -2 x_ 4 -5 x_ 5 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

\[\left\{\left[\begin{array}{c}

\end{array}\right]\right\}.\]
