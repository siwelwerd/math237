  \begin{exerciseStatement}
    Find a basis for the solution space of the homogeneous system 
\begin{align*}
 -6 x_ 4 &= 0  \\ 
  -6 x_ 1 -4 x_ 2 &= 0  \\ 
  -1 x_ 1 + 1 x_ 2 -5 x_ 4 &= 0  \\ 
  2 x_ 1 + 4 x_ 4 &= 0  \\ 
 \end{align*}


 
  \end{exerciseStatement}

  \begin{exerciseAnswer}
   A basis is   
\[\left\{\left[\begin{array}{c}
0 \\
0 \\
1 \\
0
\end{array}\right]\right\}.\]

  


  \end{exerciseAnswer}
